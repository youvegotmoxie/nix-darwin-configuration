ssh-keygen -t ed25519-sk -O resident -O application=ssh:rpi4-standalone -f .ssh/id_ed25519_standalone_sk

yubico-piv-tool -s 9a -AECCP384 -a verify-pin,generate -o public.pem
yubico-piv-tool -a verify-pin -a selfsign-certificate -s 9a --subject "/CN=SSH key/" --valid-days=3650 -i public.pem -o cert.pem
yubico-piv-tool -a import-certificate -s 9a -i cert.pem

gpg --keyserver hkps://keys.openpgp.org --send-keys $KEYID
gpg --keyserver keys.gnupg.net --send-key $KEYID
